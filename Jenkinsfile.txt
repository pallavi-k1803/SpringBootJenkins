pipeline {
    agent any

    stages {       
        stage('build'){
            steps{
                 dir ('SpringBootJenkins') {
                    withMaven {
                        bat "mvn clean"
                    }
                 }
            }
        }
        stage('Test') {
            steps {
               dir ('SpringBootJenkins') {
                    withMaven {
                        bat "mvn test"
                    }
               }
            }
        }
        stage('Deploy') {
            steps {
                dir ('SpringBootJenkins') {
                    withMaven {
                    // Run Maven on a windows agent.
                    bat "mvn install"
                    }
                }
            }
        }
    }
}
